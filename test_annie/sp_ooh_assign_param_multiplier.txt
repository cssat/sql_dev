DELIMITER $$

CREATE DEFINER=`test_annie`@`localhost` PROCEDURE `sp_ooh_assign_param_multiplier`()
BEGIN
declare i int;
set  @constant:=(select multiplier from ooh_int_hash_key where parameter_id=0);
set @mult_int_param_key:=(select multiplier from ooh_int_hash_key where parameter_id=6);
set @int_param_key=cast((@constant/@int_param_key) as unsigned);
set i=1;

while i<=16 do
        --  set  @parameter_name= (select parameter_name from ooh_int_hash_key where parameter_id=i);
      
       --  set  @multiplier= (select multiplier from ooh_int_hash_key where parameter_id=i);
        select parameter_name into @parameter_name from ooh_int_hash_key where parameter_id=i;
        set @myqry=concat('select multiplier into @mult_',@parameter_name,' from ooh_int_hash_key where parameter_id=',i);
        set i= i+ 1;

        PREPARE stmt FROM @myqry;
        EXECUTE stmt;
        DEALLOCATE PREPARE stmt;
        set @myqry='';
end while;

END
